47template<typename Key>
48class PolynomialCycleSlipDetector
49{
50 public:
54 explicit PolynomialCycleSlipDetector(size_t windowSize, size_t polyDegree)
55 : _windowSize(windowSize), _polyDegree(polyDegree) {}
56
63 [[nodiscard]] PolynomialCycleSlipDetectorResult checkForCycleSlip(const Key& key, InsTime insTime, double measurementDifference, double threshold = 0.1)
64 {
65 if (!_enabled) { return PolynomialCycleSlipDetectorResult::Disabled; }
66 if (!_detectors.contains(key))
67 {
68 addMeasurement(key, insTime, measurementDifference);
69 return PolynomialCycleSlipDetectorResult::LessDataThanWindowSize;
70 }
71
72 const auto& detector = _detectors.at(key);
73 if (!detector.polyReg.windowSizeReached())
74 {
75 addMeasurement(key, insTime, measurementDifference);
76 return PolynomialCycleSlipDetectorResult::LessDataThanWindowSize;
77 }
78
79 auto polynomial = detector.polyReg.calcPolynomial();
80 auto predictedValue = polynomial.f(calcRelativeTime(insTime, detector));
81
82 if (std::abs(measurementDifference - predictedValue) > threshold)
83 {
84 reset(key);
85 addMeasurement(key, insTime, measurementDifference);
86 return PolynomialCycleSlipDetectorResult::CycleSlip;
87 }
88 addMeasurement(key, insTime, measurementDifference);
89 return PolynomialCycleSlipDetectorResult::NoCycleSlip;
90 }
91
93 void clear()
94 {
95 _detectors.clear();
96 }

100 void reset(const Key& key)
101 {
102 if (_detectors.contains(key))
103 {
104 _detectors.erase(key);

162 private:
164 struct SignalDetector
165 {
171 SignalDetector(InsTime startTime, size_t windowSize, size_t polyDegree, Strategy strategy)
172 : startTime(startTime), polyReg(polyDegree, windowSize, strategy) {}
173
174 InsTime startTime;
175 PolynomialRegressor<double> polyReg;
176 };
177
178 bool _enabled = true;
179 size_t _windowSize;
180 size_t _polyDegree = 2;
181 Strategy _strategy = Strategy::HouseholderQR;
182 unordered_map<Key, SignalDetector> _detectors;
183
187 [[nodiscard]] static double calcRelativeTime(const InsTime& insTime, const SignalDetector& detector)
188 {
189 return static_cast<double>((insTime - detector.startTime).count());
190 }
194 [[nodiscard]] std::optional<double> calcRelativeTime(const Key& key, const InsTime& insTime) const
195 {
196 if (!_detectors.contains(key)) { return {}; }
197
198 return calcRelativeTime(insTime, _detectors.at(key));
199 }
200
204 [[nodiscard]] std::optional<double> predictValue(const Key& key, const InsTime& insTime) const
205 {
206 if (!_detectors.contains(key)) { return {}; }
207
208 const auto& detector = _detectors.at(key);
209
210 auto polynomial = detector.polyReg.calcPolynomial();
211 return polynomial.f(calcRelativeTime(insTime, detector));
212 }
213
216 [[nodiscard]] std::optional<Polynomial<double>> calcPolynomial(const Key& key) const
217 {
218 if (!_detectors.contains(key)) { return {}; }
219
220 return _detectors.at(key).polyReg.calcPolynomial();
221 }
222
227 void addMeasurement(const Key& key, InsTime insTime, double measurementDifference)
228 {
229 if (!_enabled) { return; }
230 auto& detector = _detectors.insert({ key, SignalDetector(insTime, _windowSize, _polyDegree, _strategy) }).first->second;
231
232 detector.polyReg.push_back(calcRelativeTime(insTime, detector), measurementDifference);
233 }

243template<typename Key>
244bool PolynomialCycleSlipDetectorGui(const char* label, PolynomialCycleSlipDetector<Key>& polynomialCycleSlipDetector, float width = 0.0F)
245{
246 bool changed = false;
247
248 bool enabled = polynomialCycleSlipDetector.isEnabled();
249 if (ImGui::Checkbox(fmt::format("Enabled##{}", label).c_str(), &enabled))
250 {
251 changed = true;
252 polynomialCycleSlipDetector.setEnabled(enabled);
253 }
254
255 if (!enabled) { ImGui::BeginDisabled(); }
256
257 ImGui::SetNextItemWidth(width);
258 if (int windowSize = static_cast<int>(polynomialCycleSlipDetector.getWindowSize());
259 ImGui::InputIntL(fmt::format("Window size##{}", label).c_str(), &windowSize,
260 std::max(1, static_cast<int>(polynomialCycleSlipDetector.getPolynomialDegree()) + 1)))
261 {
262 changed = true;
263 polynomialCycleSlipDetector.setWindowSize(static_cast<size_t>(windowSize));
264 }
265
266 ImGui::SetNextItemWidth(width);
267 if (int polyDegree = static_cast<int>(polynomialCycleSlipDetector.getPolynomialDegree());
268 ImGui::InputIntL(fmt::format("Polynomial Degree##{}", label).c_str(), &polyDegree,
269 0, std::min(static_cast<int>(polynomialCycleSlipDetector.getWindowSize()) - 1, std::numeric_limits<int>::max())))
270 {
271 changed = true;
272 polynomialCycleSlipDetector.setPolynomialDegree(static_cast<size_t>(polyDegree));
273 }
274
275 ImGui::SetNextItemWidth(width);
276 if (auto strategy = polynomialCycleSlipDetector.getFitStrategy();
277 gui::widgets::EnumCombo(fmt::format("Strategy##{}", label).c_str(), strategy))
278 {
279 changed = true;
280 polynomialCycleSlipDetector.setFitStrategy(strategy);
281 }
282
283 if (!enabled) { ImGui::EndDisabled(); }
284
285 return changed;
286}
