102constexpr int32_t daysInMonth(int32_t month, int32_t year)
103{
104 --month; // Make month 0 based
105 if (month >= InsTimeUtil::MONTHS_PER_YEAR)
106 {
107 year += month / InsTimeUtil::MONTHS_PER_YEAR;
108 month %= InsTimeUtil::MONTHS_PER_YEAR;
109 }
110 while (month < 0)
111 {
112 month += MONTHS_PER_YEAR;
113 year--;
114 }
115 ++month; // Make month 1 based
116
117 switch (month)
118 {
119 case 1:
120 return 31;
121 case 2:
122 if (isLeapYear(year))
123 {
124 return 29;
125 }
126 return 28;
127 case 3:
128 return 31;
129 case 4:
130 return 30;
131 case 5:
132 return 31;
133 case 6:
134 return 30;
135 case 7:
136 return 31;
137 case 8:
138 return 31;
139 case 9:
140 return 30;
141 case 10:
142 return 31;
143 case 11:
144 return 30;
145 case 12:
146 return 31;
147 default:
148 return 366;
149 }

716 + gcem::floor(static_cast<double>(y) / 400.0)
717 - 32045);

807 {
808 InsTime_MJD mjd_leap = _mjd;
809 // Convert from UTC to intended time system
810 mjd_leap.mjd_frac += static_cast<long double>(differenceToUTC(timesys)) / static_cast<long double>(InsTimeUtil::SECONDS_PER_DAY);
811
812 // Put everything in the time of week, as it gets splitted in InsTime_GPSweekTow constructor
813 auto tow = static_cast<long double>((mjd_leap.mjd_day - InsTimeUtil::DIFF_TO_6_1_1980_MJD)) * InsTimeUtil::SECONDS_PER_DAY
814 + mjd_leap.mjd_frac * InsTimeUtil::SECONDS_PER_DAY;
815
816 return { 0, 0, tow };
817 }

822 [[nodiscard]] constexpr InsTime_YMDHMS toYMDHMS(TimeSystem timesys = UTC) const
823 {
824 // transform MJD to JD
825 InsTime_JD jd = toJD();
826 jd.jd_frac = jd.jd_frac + 0.5L;
827 jd.jd_frac += static_cast<long double>(differenceToUTC(timesys)) / static_cast<long double>(InsTimeUtil::SECONDS_PER_DAY);
828 while (jd.jd_frac >= 1.0L)
829 {
830 jd.jd_day += 1;
831 jd.jd_frac -= 1.0L;
832 }
833 // transform JD to YMDHMS
834 double a = 32044.0 + jd.jd_day;
835 double b = gcem::floor((4.0 * a + 3.0) / 146097.0);
836 double c = a - gcem::floor((b * 146097.0) / 4.0);
837
838 double d = gcem::floor((4.0 * c + 3.0) / 1461.0);
839 double e = c - gcem::floor((1461.0 * d) / 4.0);
840 double m = gcem::floor((5.0 * e + 2.0) / 153.0);
841
842 auto day = static_cast<uint16_t>(e - gcem::floor((153.0 * m + 2.0) / 5.0) + 1);
843 auto month = static_cast<uint16_t>(m + 3 - 12 * gcem::floor(m / 10.0));
844 auto year = static_cast<uint16_t>(b * 100 + d - 4800.0 + gcem::floor(m / 10.0));
845
846 long double sec = jd.jd_frac * InsTimeUtil::SECONDS_PER_DAY;
847
848 return { year, month, day, 0, 0, sec };
849 }

854 [[nodiscard]] constexpr InsTime_YDoySod toYDoySod(TimeSystem timesys = UTC) const
855 {
856 InsTime_YMDHMS yearMonthDayHMS = toYMDHMS();
857
858 auto year = yearMonthDayHMS.year;
859 long double sod = static_cast<long double>(yearMonthDayHMS.hour * InsTimeUtil::SECONDS_PER_HOUR
860 + yearMonthDayHMS.min * InsTimeUtil::SECONDS_PER_MINUTE)
861 + yearMonthDayHMS.sec
862 + static_cast<long double>(differenceToUTC(timesys));
863
864 int32_t doy = 0;
865 for (int32_t i = 1; i < yearMonthDayHMS.month; i++)
866 {
867 doy += InsTimeUtil::daysInMonth(i, year);
868 }
869 doy += yearMonthDayHMS.day;
870
871 return { year, doy, sod };
872 }
