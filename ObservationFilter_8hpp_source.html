117 template<typename ReceiverType>
118 [[nodiscard]] Observations selectObservationsForCalculation(const std::array<Receiver<ReceiverType>, ReceiverType::ReceiverType_COUNT>& receivers,
119 const std::vector<const GnssNavInfo*>& gnssNavInfos,
120 [[maybe_unused]] const std::string& nameId,
121 bool ignoreElevationMask = false)
122 {
123 Observations observations;
124 observations.signals.reserve(receivers.front().gnssObs->data.size());
125
126 std::array<std::unordered_set<SatId>, GnssObs::ObservationType_COUNT> nMeasUniqueSat;
127
128 for (const auto& obsData : receivers.front().gnssObs->data)
129 {
130 SatId satId = obsData.satSigId.toSatId();
131
132 if (!(obsData.satSigId.freq() & _filterFreq) // frequency is not selected in GUI
133 || !(obsData.satSigId.code & _filterCode) // code is not selected in GUI
134 || !obsData.pseudorange // has an invalid pseudorange
135 || std::find(_excludedSatellites.begin(), _excludedSatellites.end(), satId) != _excludedSatellites.end()) // is excluded
136 {
137 LOG_DATA("{}: [{}] Skipping obs due to GUI filter selections", nameId, obsData.satSigId);
138 continue;
139 }
140
141 LOG_DATA("{}: [{}] Searching if observation in all receivers", nameId, obsData.satSigId);
142 unordered_map<GnssObs::ObservationType, size_t> availableObservations;
143 if (std::any_of(receivers.begin(), receivers.end(),
144 [&](const Receiver<ReceiverType>& recv) {
145 auto obsDataOther = std::find_if(recv.gnssObs->data.begin(), recv.gnssObs->data.end(),
146 [&obsData](const GnssObs::ObservationData& obsDataOther) {
147 return obsDataOther.satSigId == obsData.satSigId;
148 });
149
150 if (obsDataOther == recv.gnssObs->data.end())
151 {
152 LOG_DATA("{}: [{}] Receiver '{}' did not observe the signal.", nameId, obsData.satSigId, recv.type);
153 return true;
154 } // All receivers must have this observation
155
156 for (const auto& obsType : _usedObsTypes)
157 {
158 switch (obsType)
159 {
160 case GnssObs::Pseudorange:
161 if (obsDataOther->pseudorange) { availableObservations[obsType]++; }
162 break;
163 case GnssObs::Carrier:
164 if (obsDataOther->carrierPhase) { availableObservations[obsType]++; }
165 break;
166 case GnssObs::Doppler:
167 if (obsDataOther->doppler) { availableObservations[obsType]++; }
168 break;
169 case GnssObs::ObservationType_COUNT:
170 break;
171 }
172 }
173 if (!obsDataOther->pseudorange)
174 {
175 LOG_DATA("{}: [{}] Receiver '{}' did not have a pseudorange observation.", nameId, obsData.satSigId, recv.type);
176 }
177 return !obsDataOther->pseudorange; // Pseudorange must be available for all receivers
178 }))
179 {
180 continue;
181 }
182 LOG_DATA("{}: [{}] Observed psr {}x, carrier {}x, doppler {}x", nameId, obsData.satSigId,
183 availableObservations.contains(GnssObs::Pseudorange) ? availableObservations.at(GnssObs::Pseudorange) : 0,
184 availableObservations.contains(GnssObs::Carrier) ? availableObservations.at(GnssObs::Carrier) : 0,
185 availableObservations.contains(GnssObs::Doppler) ? availableObservations.at(GnssObs::Doppler) : 0);
186
187 std::shared_ptr<NAV::SatNavData> satNavData = nullptr;
188 for (const auto& gnssNavInfo : gnssNavInfos)
189 {
190 auto satNav = gnssNavInfo->searchNavigationData(satId, receivers.front().gnssObs->insTime);
191 if (satNav && satNav->isHealthy())
192 {
193 satNavData = satNav;
194 break;
195 }
196 }
197 if (satNavData == nullptr) { continue; } // can calculate satellite position
198
199 int8_t freqNum = -128;
200 if (satId.satSys == GLO)
201 {
202 if (auto gloSatNavData = std::dynamic_pointer_cast<GLONASSEphemeris>(satNavData))
203 {
204 freqNum = gloSatNavData->frequencyNumber;
205 }
206 }
207
208 Observations::SignalObservation sigObs(satNavData, freqNum);
209
210 bool skipObservation = false;
211 for (const auto& recv : receivers)
212 {
213 auto recvObsData = std::find_if(recv.gnssObs->data.begin(), recv.gnssObs->data.end(),
214 [&obsData](const GnssObs::ObservationData& recvObsData) {
215 return recvObsData.satSigId == obsData.satSigId;
216 });
217 auto satClk = satNavData->calcClockCorrections(recv.gnssObs->insTime,
218 recvObsData->pseudorange->value,
219 recvObsData->satSigId.freq());
220 auto satPosVel = satNavData->calcSatellitePosVel(satClk.transmitTime);
221
222 LOG_DATA("{}: Adding satellite [{}] for receiver {}", nameId, obsData.satSigId, recv.type);
223 sigObs.recvObs.emplace_back(recv.gnssObs, static_cast<size_t>(recvObsData - recv.gnssObs->data.begin()),
224 recv.e_pos, recv.lla_pos, recv.e_vel,
225 satPosVel.e_pos, satPosVel.e_vel, satClk);
226
227 if (!ignoreElevationMask)
228 {
229 const auto& satElevation = sigObs.recvObs.back().satElevation();
230 if (satElevation < _elevationMask)
231 {
232 LOG_DATA("{}: Signal {} is skipped because of elevation mask. ({} < {})", nameId, obsData.satSigId,
233 rad2deg(satElevation), rad2deg(_elevationMask));
234 skipObservation = true;
235 break;
236 }
237 if (recvObsData->CN0 // If no CN0 available, we use the signal
238 && !_snrMask
239 .at(_sameSnrMaskForAllReceivers ? static_cast<ReceiverType>(0) : recv.type)
240 .checkSNRMask(obsData.satSigId.freq(), satElevation, recvObsData->CN0.value()))
241 {
242 LOG_DEBUG("{}: [{}] SNR mask triggered for [{}] on receiver [{}] with CN0 {} dbHz",
243 nameId, receivers.front().gnssObs->insTime.toYMDHMS(GPST), obsData.satSigId, recv.type, recvObsData->CN0.value());
244 skipObservation = true;
245 break;
246 }
247 }
248 }
249 if (skipObservation) { continue; }
250
251 for (const auto& recv : receivers)
252 {
253 auto& recvObsData = sigObs.recvObs.at(recv.type);
254
255 for (const auto& obsType : _usedObsTypes)
256 {
257 if (availableObservations.contains(obsType) && availableObservations.at(obsType) == receivers.size())
258 {
259 observations.nObservables.at(obsType)++;
260 nMeasUniqueSat.at(obsType).insert(satId);
261 switch (obsType)
262 {
263 case GnssObs::Pseudorange:
264 recvObsData.obs[obsType].measurement = recvObsData.gnssObsData().pseudorange->value;
265 LOG_DATA("{}: [{}] Taking {:11} observation into account on {:5} receiver ({:.3f} [m])", nameId, obsData.satSigId,
266 obsType, recv.type, recvObsData.obs[obsType].measurement);
267 break;
268 case GnssObs::Carrier:
269 recvObsData.obs[obsType].measurement = InsConst<>::C / obsData.satSigId.freq().getFrequency(freqNum)
270 * recvObsData.gnssObsData().carrierPhase->value;
271 LOG_DATA("{}: [{}] Taking {:11} observation into account on {:5} receiver ({:.3f} [m] = {:.3f} [cycles])", nameId, obsData.satSigId,
272 obsType, recv.type, recvObsData.obs[obsType].measurement, recvObsData.gnssObsData().carrierPhase->value);
273 break;
274 case GnssObs::Doppler:
275 recvObsData.obs[obsType].measurement = doppler2rangeRate(recvObsData.gnssObsData().doppler.value(),
276 obsData.satSigId.freq(),
277 freqNum);
278 LOG_DATA("{}: [{}] Taking {:11} observation into account on {:5} receiver ({:.3f} [m/s] = {:.3f} [Hz])", nameId, obsData.satSigId,
279 obsType, recv.type, recvObsData.obs[obsType].measurement, recvObsData.gnssObsData().doppler.value());
280 break;
281 case GnssObs::ObservationType_COUNT:
282 break;
283 }
284 }
285 }
286 }
287
288 observations.systems.insert(satId.satSys);
289 observations.satellites.insert(satId);
290 observations.signals.insert(std::make_pair(obsData.satSigId, sigObs));
291 }
292
293 for (size_t obsType = 0; obsType < GnssObs::ObservationType_COUNT; obsType++)
294 {
295 observations.nObservablesUniqueSatellite.at(obsType) = nMeasUniqueSat.at(obsType).size();
296 }
297
298#if LOG_LEVEL <= LOG_LEVEL_DATA
299 LOG_DATA("{}: usedSatSystems = [{}]", nameId, joinToString(observations.systems));
300 size_t nMeasTotal = 0;
301 std::string nMeasStr;
302 for (size_t obsType = 0; obsType < GnssObs::ObservationType_COUNT; obsType++)
303 {
304 auto& nMeas = observations.nObservables.at(obsType);
305 nMeas /= receivers.size();
306 nMeasStr += fmt::format("{} {}, ", nMeas, static_cast<GnssObs::ObservationType>(obsType));
307 nMeasTotal += nMeas;
308 }
309 if (nMeasStr.ends_with(", ")) { nMeasStr = nMeasStr.erase(nMeasStr.length() - 2); }
310
311 LOG_DATA("{}: Using {} measurements ({}) from {} satellites", nameId, nMeasTotal, nMeasStr, observations.satellites.size());
312
313 unordered_map<SatId, std::pair<Frequency, Code>> satData;
314 for (const auto& obs : observations.signals)
315 {
316 satData[obs.first.toSatId()].first |= obs.first.freq();
317 satData[obs.first.toSatId()].second |= obs.first.code;
318 }
319 for (const auto& [satId, freqCode] : satData)
320 {
321 LOG_DATA("{}: [{}] on frequencies [{}] with codes [{}]", nameId, satId, freqCode.first, freqCode.second);
322 }
323#endif
324
325 return observations;
326 }

331 template<typename ReceiverType>
332 bool ShowGuiWidgets(const char* id, float itemWidth)
333 {
334 bool changed = false;
335
336 ImGui::SetNextItemWidth(itemWidth);
337 if (ShowFrequencySelector(fmt::format("Satellite Frequencies##{}", id).c_str(), _filterFreq))
338 {
339 changed = true;
340 }
341
342 ImGui::SetNextItemWidth(itemWidth);
343 if (ShowCodeSelector(fmt::format("Signal Codes##{}", id).c_str(), _filterCode, _filterFreq))
344 {
345 changed = true;
346 }
347
348 ImGui::SetNextItemWidth(itemWidth);
349 if (ShowSatelliteSelector(fmt::format("Excluded satellites##{}", id).c_str(), _excludedSatellites))
350 {
351 changed = true;
352 }
353
354 double elevationMaskDeg = rad2deg(_elevationMask);
355 ImGui::SetNextItemWidth(itemWidth);
356 if (ImGui::InputDoubleL(fmt::format("Elevation mask##{}", id).c_str(), &elevationMaskDeg, 0.0, 90.0, 5.0, 5.0, "%.1f", ImGuiInputTextFlags_AllowTabInput))
357 {
358 _elevationMask = deg2rad(elevationMaskDeg);
359 LOG_DEBUG("{}: Elevation mask changed to {}", id, elevationMaskDeg);
360 changed = true;
361 }
362
363 for (size_t i = 0; i < ReceiverType::ReceiverType_COUNT; ++i)
364 {
365 if (i != 0)
366 {
367 ImGui::SameLine();
368 if (_sameSnrMaskForAllReceivers) { ImGui::BeginDisabled(); }
369 }
370 if (_snrMask.at(i).ShowGuiWidgets(fmt::format("{} SNR Mask", static_cast<ReceiverType>(i)).c_str()))
371 {
372 changed = true;
373 }
374 if (i != 0 && _sameSnrMaskForAllReceivers) { ImGui::EndDisabled(); }
375 }
376 if (ReceiverType::ReceiverType_COUNT > 1)
377 {
378 ImGui::SameLine();
379 if (ImGui::Checkbox(fmt::format("Use same SNR for all receivers##{}", id).c_str(), &_sameSnrMaskForAllReceivers))
380 {
381 changed = true;
382 }
383 }
384
385 ImGui::BeginHorizontal(fmt::format("Observables##{}", id).c_str(),
386 ImVec2(itemWidth - ImGui::GetStyle().ItemSpacing.x + ImGui::GetStyle().ItemInnerSpacing.x, 0.0F));
387 for (size_t i = 0; i < GnssObs::ObservationType_COUNT; i++)
388 {
389 auto obsType = static_cast<GnssObs::ObservationType>(i);
390 if (!_availableObsTypes.contains(obsType)) { continue; }
391 if (_neededObsTypes.contains(obsType)) { ImGui::BeginDisabled(); }
392 bool enabled = _usedObsTypes.contains(obsType);
393 if (ImGui::Checkbox(fmt::format("{}##{}", obsType, id).c_str(), &enabled))
394 {
395 LOG_DEBUG("{}: Using {}: {}", id, obsType, enabled);
396 if (enabled) { _usedObsTypes.insert(obsType); }
397 else { _usedObsTypes.erase(obsType); }
398 changed = true;
399 }
400 if (_neededObsTypes.contains(obsType)) { ImGui::EndDisabled(); }
401 }
402 ImGui::EndHorizontal();
403
404 ImGui::SameLine();
405 ImGui::TextUnformatted("Used observables");
406
407 return changed;
408 }
