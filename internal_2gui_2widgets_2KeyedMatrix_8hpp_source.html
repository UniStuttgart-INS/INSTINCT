33template<typename Scalar, typename RowKeyType, typename ColKeyType, int Rows, int Cols>
34void KeyedMatrixView(const char* label, const KeyedMatrix<Scalar, RowKeyType, ColKeyType, Rows, Cols>* matrix,
35 float tableHeight = -1.0F,
36 ImGuiTableFlags tableFlags = ImGuiTableFlags_Borders
37 | ImGuiTableFlags_NoHostExtendX
38 | ImGuiTableFlags_SizingFixedFit
39 | ImGuiTableFlags_ScrollX
40 | ImGuiTableFlags_ScrollY)
41{
42 ImGui::PushFont(Application::MonoFont());
43 float height = ImGui::GetTextLineHeightWithSpacing() * static_cast<float>(matrix->rows() + 1);
44 ImVec2 outer_size = ImVec2(0.0F, tableHeight > 0.0F ? std::min(tableHeight, height) : height);
45 if (ImGui::BeginTable(label, static_cast<int>(matrix->cols()) + 1, tableFlags, outer_size))
46 {
47 ImGui::TableSetupScrollFreeze(1, 1);
48 ImGui::TableSetupColumn(""); // Colum headers
49
50 constexpr size_t colMinLength = 10UL;
51 std::vector<size_t> colKeysLength;
52 colKeysLength.reserve(static_cast<size_t>(matrix->cols()));
53
54 for (int64_t col = 0; col < matrix->cols(); col++)
55 {
56 std::string colKeyStr = fmt::format("{}", matrix->colKeys().at(static_cast<size_t>(col)));
57 colKeysLength.push_back(colKeyStr.length());
58 ImGui::TableSetupColumn(colKeyStr.c_str());
59 }
60 ImGui::TableHeadersRow();
61
62 for (int64_t row = 0; row < matrix->rows(); row++)
63 {
64 ImGui::TableNextColumn();
65 ImGui::TextUnformatted(fmt::format("{}", matrix->rowKeys().at(static_cast<size_t>(row))).c_str());
66 ImU32 cell_bg_color = ImGui::GetColorU32(ImGui::GetStyle().Colors[ImGuiCol_TableHeaderBg]);
67 ImGui::TableSetBgColor(ImGuiTableBgTarget_CellBg, cell_bg_color);
68
69 for (int64_t col = 0; col < matrix->cols(); col++)
70 {
71 ImGui::TableNextColumn();
72
73 auto colLength = std::max(colKeysLength.at(static_cast<size_t>(col)), colMinLength);
74 std::string text = fmt::format(" {:> {}.{}g}", (*matrix)(NAV::all, NAV::all)(row, col), colLength, colLength - 2);
75 if (text.length() > colLength)
76 {
77 text = fmt::format(" {:> {}.{}g}", (*matrix)(NAV::all, NAV::all)(row, col), colLength, colLength - 6);
78 }
79 ImGui::TextUnformatted(text.c_str());
80 if (ImGui::IsItemHovered())
81 {
82 ImGui::SetTooltip("%.8g", (*matrix)(NAV::all, NAV::all)(row, col));
83 }
84 }
85 }
86 ImGui::EndTable();
87 }
88 ImGui::PopFont();
89}

213template<typename Scalar, typename RowKeyType, typename ColKeyType, int Rows, int Cols>
214bool InputKeyedMatrix(const char* label, KeyedMatrix<Scalar, RowKeyType, ColKeyType, Rows, Cols>* matrix,
215 float tableHeight = -1.0F,
216 ImGuiTableFlags tableFlags = ImGuiTableFlags_Borders
217 | ImGuiTableFlags_NoHostExtendX
218 | ImGuiTableFlags_SizingFixedFit
219 | ImGuiTableFlags_ScrollX
220 | ImGuiTableFlags_ScrollY,
221 double step = 0.0, double step_fast = 0.0,
222 const char* format = "%.4g",
223 ImGuiInputTextFlags inputTextFlags = ImGuiInputTextFlags_None)
224{
225 bool changed = false;
226 ImGui::PushFont(Application::MonoFont());
227 const float TEXT_BASE_WIDTH = ImGui::CalcTextSize("A").x;
228 float height = ImGui::GetTextLineHeightWithSpacing()
229 + (ImGui::GetTextLineHeightWithSpacing() + 1.5F * ImGui::GetStyle().ItemSpacing.y)
230 * static_cast<float>(matrix->rows());
231 ImVec2 outer_size = ImVec2(0.0F, tableHeight > 0.0F ? std::min(tableHeight, height) : height);
232 if (ImGui::BeginTable(label, static_cast<int>(matrix->cols()) + 1, tableFlags, outer_size))
233 {
234 ImGui::TableSetupScrollFreeze(1, 1);
235 ImGui::TableSetupColumn(""); // Colum headers
236
237 constexpr size_t colMinLength = 11UL;
238 std::vector<size_t> colKeysLength;
239 colKeysLength.reserve(static_cast<size_t>(matrix->cols()));
240
241 for (int64_t col = 0; col < matrix->cols(); col++)
242 {
243 std::string colKeyStr = fmt::format("{}", matrix->colKeys().at(static_cast<size_t>(col)));
244 colKeysLength.push_back(colKeyStr.length());
245 ImGui::TableSetupColumn(colKeyStr.c_str());
246 }
247 ImGui::TableHeadersRow();
248
249 for (int64_t row = 0; row < matrix->rows(); row++)
250 {
251 ImGui::TableNextColumn();
252 ImGui::TextUnformatted(fmt::format("{}", matrix->rowKeys().at(static_cast<size_t>(row))).c_str());
253 ImU32 cell_bg_color = ImGui::GetColorU32(ImGui::GetStyle().Colors[ImGuiCol_TableHeaderBg]);
254 ImGui::TableSetBgColor(ImGuiTableBgTarget_CellBg, cell_bg_color);
255
256 for (int64_t col = 0; col < matrix->cols(); col++)
257 {
258 ImGui::TableNextColumn();
259
260 auto colLength = std::max(colKeysLength.at(static_cast<size_t>(col)), colMinLength);
261 ImGui::SetNextItemWidth(TEXT_BASE_WIDTH * static_cast<float>(colLength));
262 if (ImGui::InputDouble(fmt::format("##{} ({}, {})", label, row, col).c_str(), &(*matrix)(NAV::all, NAV::all)(row, col), step, step_fast, format, inputTextFlags))
263 {
264 changed = true;
265 }
266 if (ImGui::IsItemHovered())
267 {
268 ImGui::SetTooltip("%.8g", (*matrix)(NAV::all, NAV::all)(row, col));
269 }
270 }
271 }
272 ImGui::EndTable();
273 }
274 ImGui::PopFont();
275
276 return changed;
