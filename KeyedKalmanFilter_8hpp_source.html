41 KeyedKalmanFilter(const std::vector<StateKeyType>& stateKeys, const std::vector<MeasKeyType>& measKeys)
42 {
43 std::unordered_set<StateKeyType> stateSet = { stateKeys.begin(), stateKeys.end() };
44 INS_ASSERT_USER_ERROR(stateSet.size() == stateKeys.size(), "Each state key must be unique");
45 std::unordered_set<MeasKeyType> measSet = { measKeys.begin(), measKeys.end() };
46 INS_ASSERT_USER_ERROR(measSet.size() == measKeys.size(), "Each measurement key must be unique");
47
48 auto n = static_cast<int>(stateKeys.size());
49 auto m = static_cast<int>(measKeys.size());
50
51 x = KeyedVectorX<Scalar, StateKeyType>(Eigen::VectorX<Scalar>::Zero(n), stateKeys);
52 P = KeyedMatrixX<Scalar, StateKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(n, n), stateKeys);
53 F = KeyedMatrixX<Scalar, StateKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(n, n), stateKeys);
54 Phi = KeyedMatrixX<Scalar, StateKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(n, n), stateKeys);
55
56 G = KeyedMatrixX<Scalar, StateKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(n, n), stateKeys);
57 W = KeyedMatrixX<Scalar, StateKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(n, n), stateKeys);
58 Q = KeyedMatrixX<Scalar, StateKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(n, n), stateKeys);
59 z = KeyedVectorX<Scalar, MeasKeyType>(Eigen::VectorX<Scalar>::Zero(m), measKeys);
60 H = KeyedMatrixX<Scalar, MeasKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(m, n), measKeys, stateKeys);
61 R = KeyedMatrixX<Scalar, MeasKeyType, MeasKeyType>(Eigen::MatrixX<Scalar>::Zero(m, m), measKeys, measKeys);
62 S = KeyedMatrixX<Scalar, MeasKeyType, MeasKeyType>(Eigen::MatrixX<Scalar>::Zero(m, m), measKeys, measKeys);
63 K = KeyedMatrixX<Scalar, StateKeyType, MeasKeyType>(Eigen::MatrixX<Scalar>::Zero(n, m), stateKeys, measKeys);
64 I = Eigen::MatrixX<Scalar>::Identity(n, n);
65 }

87 void predict()
88 {
89 // Math: \mathbf{\hat{x}}_k^- = \mathbf{\Phi}_{k-1}\mathbf{\hat{x}}_{k-1}^+ \qquad \text{P. Groves}\,(3.14)
90 x(all) = Phi(all, all) * x(all);
91
92 // Math: \mathbf{P}_k^- = \mathbf{\Phi}_{k-1} P_{k-1}^+ \mathbf{\Phi}_{k-1}^T + \mathbf{Q}_{k-1} \qquad \text{P. Groves}\,(3.15)
93 P(all, all) = Phi(all, all) * P(all, all) * Phi(all, all).transpose() + Q(all, all);
94 }
95
100 void correct()
101 {
102 S(all, all) = H(all, all) * P(all, all) * H(all, all).transpose() + R(all, all);
103
104 // Math: \mathbf{K}_k = \mathbf{P}_k^- \mathbf{H}_k^T (\mathbf{H}_k \mathbf{P}_k^- \mathbf{H}_k^T + R_k)^{-1} \qquad \text{P. Groves}\,(3.21)
105 K(all, all) = P(all, all) * H(all, all).transpose() * S(all, all).inverse();
106
107 // Math: \begin{align*} \mathbf{\hat{x}}_k^+ &= \mathbf{\hat{x}}_k^- + \mathbf{K}_k (\mathbf{z}_k - \mathbf{H}_k \mathbf{\hat{x}}_k^-) \\ &= \mathbf{\hat{x}}_k^- + \mathbf{K}_k \mathbf{\delta z}_k^{-} \end{align*} \qquad \text{P. Groves}\,(3.24)
108 x(all) = x(all) + K(all, all) * (z(all) - H(all, all) * x(all));
109
110 // Math: \mathbf{P}_k^+ = (\mathbf{I} - \mathbf{K}_k \mathbf{H}_k) \mathbf{P}_k^- \qquad \text{P. Groves}\,(3.25)
111 P(all, all) = (I - K(all, all) * H(all, all)) * P(all, all);
112 }
113
119 void correctWithMeasurementInnovation()
120 {
121 S(all, all) = H(all, all) * P(all, all) * H(all, all).transpose() + R(all, all);
122
123 // Math: \mathbf{K}_k = \mathbf{P}_k^- \mathbf{H}_k^T (\mathbf{H}_k \mathbf{P}_k^- \mathbf{H}_k^T + R_k)^{-1} \qquad \text{P. Groves}\,(3.21)
124 K(all, all) = P(all, all) * H(all, all).transpose() * S(all, all).inverse();
125
126 // Math: \begin{align*} \mathbf{\hat{x}}_k^+ &= \mathbf{\hat{x}}_k^- + \mathbf{K}_k (\mathbf{z}_k - \mathbf{H}_k \mathbf{\hat{x}}_k^-) \\ &= \mathbf{\hat{x}}_k^- + \mathbf{K}_k \mathbf{\delta z}_k^{-} \end{align*} \qquad \text{P. Groves}\,(3.24)
127 x(all) = x(all) + K(all, all) * z(all);
128
129 // Math: \mathbf{P}_k^+ = (\mathbf{I} - \mathbf{K}_k \mathbf{H}_k) \mathbf{P}_k^- \qquad \text{P. Groves}\,(3.25)
130 // P(all, all) = (I - K(all, all) * H(all, all)) * P(all, all);
131
132 // Math: \mathbf{P}_k^+ = (\mathbf{I} - \mathbf{K}_k \mathbf{H}_k) \mathbf{P}_k^- (\mathbf{I} - \mathbf{K}_k \mathbf{H}_k)^T + \mathbf{K}_k \mathbf{R}_k \mathbf{K}_k^T \qquad \text{Brown & Hwang}\,(p. 145, eq. 4.2.11)
133 P(all, all) = (I - K(all, all) * H(all, all)) * P(all, all) * (I - K(all, all) * H(all, all)).transpose() + K(all, all) * R(all, all) * K(all, all).transpose();
134 }

376template<typename Scalar, typename RowKeyType, typename ColKeyType, int Rows, int Cols>
377KeyedMatrix<Scalar, RowKeyType, ColKeyType, Rows, Cols> transitionMatrix_Phi_Taylor(const KeyedMatrix<Scalar, RowKeyType, ColKeyType, Rows, Cols>& F, Scalar tau_s, size_t order)
378{
379 // Transition matrix ùöΩ
380 Eigen::Matrix<Scalar, Rows, Cols> Phi;
381
382 if constexpr (Rows == Eigen::Dynamic)
383 {
384 Phi = Eigen::Matrix<Scalar, Rows, Cols>::Identity(F.rows(), F.cols());
385 }
386 else
387 {
388 Phi = Eigen::Matrix<Scalar, Rows, Cols>::Identity();
389 }
390 // std::cout << "Phi = I";
391
392 for (size_t i = 1; i <= order; i++)
393 {
394 Eigen::Matrix<Scalar, Rows, Cols> Fpower = F(all, all);
395 // std::cout << " + (F";
396
397 for (size_t j = 1; j < i; j++) // F^j
398 {
399 // std::cout << "*F";
400 Fpower *= F(all, all);
401 }
402 // std::cout << "*tau_s^" << i << ")";
403 // std::cout << "/" << math::factorial(i);
404 Phi += (Fpower * std::pow(tau_s, i)) / math::factorial(i);
405 }
406 // std::cout << "\n";
407
408 return { Phi, F.rowKeys(), F.colKeys() };
409}
