38 KeyedKalmanFilter(const std::vector<StateKeyType>& stateKeys, const std::vector<MeasKeyType>& measKeys)
39 {
40 std::unordered_set<StateKeyType> stateSet = { stateKeys.begin(), stateKeys.end() };
41 INS_ASSERT_USER_ERROR(stateSet.size() == stateKeys.size(), "Each state key must be unique");
42 std::unordered_set<MeasKeyType> measSet = { measKeys.begin(), measKeys.end() };
43 INS_ASSERT_USER_ERROR(measSet.size() == measKeys.size(), "Each measurement key must be unique");
44
45 auto n = static_cast<int>(stateKeys.size());
46 auto m = static_cast<int>(measKeys.size());
47
48 x = KeyedVectorX<Scalar, StateKeyType>(Eigen::VectorX<Scalar>::Zero(n), stateKeys);
49 P = KeyedMatrixX<Scalar, StateKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(n, n), stateKeys);
50 Phi = KeyedMatrixX<Scalar, StateKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(n, n), stateKeys);
51 Q = KeyedMatrixX<Scalar, StateKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(n, n), stateKeys);
52 z = KeyedVectorX<Scalar, MeasKeyType>(Eigen::VectorX<Scalar>::Zero(m), measKeys);
53 H = KeyedMatrixX<Scalar, MeasKeyType, StateKeyType>(Eigen::MatrixX<Scalar>::Zero(m, n), measKeys, stateKeys);
54 R = KeyedMatrixX<Scalar, MeasKeyType, MeasKeyType>(Eigen::MatrixX<Scalar>::Zero(m, m), measKeys, measKeys);
55 S = KeyedMatrixX<Scalar, MeasKeyType, MeasKeyType>(Eigen::MatrixX<Scalar>::Zero(m, m), measKeys, measKeys);
56 K = KeyedMatrixX<Scalar, StateKeyType, MeasKeyType>(Eigen::MatrixX<Scalar>::Zero(n, m), stateKeys, measKeys);
57 I = Eigen::MatrixX<Scalar>::Identity(n, n);
58 }

77 void predict()
78 {
79 // Math: \mathbf{\hat{x}}_k^- = \mathbf{\Phi}_{k-1}\mathbf{\hat{x}}_{k-1}^+ \qquad \text{P. Groves}\,(3.14)
80 x(all) = Phi(all, all) * x(all);
81
82 // Math: \mathbf{P}_k^- = \mathbf{\Phi}_{k-1} P_{k-1}^+ \mathbf{\Phi}_{k-1}^T + \mathbf{Q}_{k-1} \qquad \text{P. Groves}\,(3.15)
83 P(all, all) = Phi(all, all) * P(all, all) * Phi(all, all).transpose() + Q(all, all);
84 }
85
90 void correct()
91 {
92 S(all, all) = H(all, all) * P(all, all) * H(all, all).transpose() + R(all, all);
93
94 // Math: \mathbf{K}_k = \mathbf{P}_k^- \mathbf{H}_k^T (\mathbf{H}_k \mathbf{P}_k^- \mathbf{H}_k^T + R_k)^{-1} \qquad \text{P. Groves}\,(3.21)
95 K(all, all) = P(all, all) * H(all, all).transpose() * S(all, all).inverse();
96
97 // Math: \begin{align*} \mathbf{\hat{x}}_k^+ &= \mathbf{\hat{x}}_k^- + \mathbf{K}_k (\mathbf{z}_k - \mathbf{H}_k \mathbf{\hat{x}}_k^-) \\ &= \mathbf{\hat{x}}_k^- + \mathbf{K}_k \mathbf{\delta z}_k^{-} \end{align*} \qquad \text{P. Groves}\,(3.24)
98 x(all) = x(all) + K(all, all) * (z(all) - H(all, all) * x(all));
99
100 // Math: \mathbf{P}_k^+ = (\mathbf{I} - \mathbf{K}_k \mathbf{H}_k) \mathbf{P}_k^- \qquad \text{P. Groves}\,(3.25)
101 P(all, all) = (I - K(all, all) * H(all, all)) * P(all, all);
102 }
103
109 void correctWithMeasurementInnovation()
110 {
111 // Math: \mathbf{K}_k = \mathbf{P}_k^- \mathbf{H}_k^T (\mathbf{H}_k \mathbf{P}_k^- \mathbf{H}_k^T + R_k)^{-1} \qquad \text{P. Groves}\,(3.21)
112 K(all, all) = P(all, all) * H(all, all).transpose() * (H(all, all) * P(all, all) * H(all, all).transpose() + R(all, all)).inverse();
113
114 // Math: \begin{align*} \mathbf{\hat{x}}_k^+ &= \mathbf{\hat{x}}_k^- + \mathbf{K}_k (\mathbf{z}_k - \mathbf{H}_k \mathbf{\hat{x}}_k^-) \\ &= \mathbf{\hat{x}}_k^- + \mathbf{K}_k \mathbf{\delta z}_k^{-} \end{align*} \qquad \text{P. Groves}\,(3.24)
115 x(all) = x(all) + K(all, all) * z(all);
116
117 // Math: \mathbf{P}_k^+ = (\mathbf{I} - \mathbf{K}_k \mathbf{H}_k) \mathbf{P}_k^- \qquad \text{P. Groves}\,(3.25)
118 // _P(all, all) = (I - K(all, all) * H(all, all)) * _P(all, all);
119
120 // Math: \mathbf{P}_k^+ = (\mathbf{I} - \mathbf{K}_k \mathbf{H}_k) \mathbf{P}_k^- (\mathbf{I} - \mathbf{K}_k \mathbf{H}_k)^T + \mathbf{K}_k \mathbf{R}_k \mathbf{K}_k^T \qquad \text{Brown & Hwang}\,(p. 145, eq. 4.2.11)
121 P(all, all) = (I - K(all, all) * H(all, all)) * P(all, all) * (I - K(all, all) * H(all, all)).transpose() + K(all, all) * R(all, all) * K(all, all).transpose();
122 }

128 static KeyedMatrixXd<StateKeyType> calcTransitionMatrix(const KeyedMatrixXd<StateKeyType>& F, double tau_s)
129 {
130 // Transition matrix ùöΩ
131 return { Eigen::MatrixXd::Identity(F(all, all).rows(), F(all, all).cols()) + F(all, all) * tau_s, F.rowKeys() };
132 }

140 void addStates(const std::vector<StateKeyType>& stateKeys)
141 {
142 INS_ASSERT_USER_ERROR(!x.hasAnyRows(stateKeys), "You cannot add a state key which is already in the Kalman filter.");

224template<typename Scalar, typename RowKeyType, typename ColKeyType, int Rows, int Cols>
225KeyedMatrix<Scalar, RowKeyType, ColKeyType, Rows, Cols> transitionMatrix_Phi_Taylor(const KeyedMatrix<Scalar, RowKeyType, ColKeyType, Rows, Cols>& F, Scalar tau_s, size_t order)
226{
227 // Transition matrix ùöΩ
228 Eigen::Matrix<Scalar, Rows, Cols> Phi;
229
230 if constexpr (Rows == Eigen::Dynamic)
231 {
232 Phi = Eigen::Matrix<Scalar, Rows, Cols>::Identity(F.rows(), F.cols());
233 }
234 else
235 {
236 Phi = Eigen::Matrix<Scalar, Rows, Cols>::Identity();
237 }
238 // std::cout << "Phi = I";
239
240 for (size_t i = 1; i <= order; i++)
241 {
242 Eigen::Matrix<Scalar, Rows, Cols> Fpower = F(all, all);
243 // std::cout << " + (F";
244
245 for (size_t j = 1; j < i; j++) // F^j
246 {
247 // std::cout << "*F";
248 Fpower *= F(all, all);
249 }
250 // std::cout << "*tau_s^" << i << ")";
251 // std::cout << "/" << math::factorial(i);
252 Phi += (Fpower * std::pow(tau_s, i)) / math::factorial(i);
253 }
254 // std::cout << "\n";
255
256 return { Phi, F.rowKeys(), F.colKeys() };
257}
