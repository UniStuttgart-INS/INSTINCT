41template<typename Y, typename Scalar,
42 typename = std::enable_if_t<std::is_floating_point_v<Scalar>>>
43Y Heun(const auto& f, const Scalar& h, const Y& n_y, const Scalar& n_t, const auto& c)
44{
45 Y y_n_p1 = n_y + h * f(n_y, n_t, c);

130template<typename Y, typename Scalar,
131 typename = std::enable_if_t<std::is_floating_point_v<Scalar>>>
132Y RungeKutta3(const auto& f, const Scalar& h, const Y& n_y, const Scalar& n_t, const auto& c)
133{
134 Y k1 = f(n_y, n_t, c);
135 Y k2 = f(n_y + h / 2 * k1, n_t + h / 2, c);
136 Y k3 = f(n_y - h * k1 + 2 * h * k2, n_t + h, c);
137 return n_y + h / 6 * (k1 + 4 * k2 + k3);
138}
