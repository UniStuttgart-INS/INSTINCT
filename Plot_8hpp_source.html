179 void addPlotDataItem(size_t dataIndex, const std::string& displayName)
180 {
181 if (plotData.size() > dataIndex)
182 {
183 if (plotData.at(dataIndex).displayName == displayName) // Item was restored already at this position
184 {
185 plotData.at(dataIndex).markedForDelete = false;
186 return;
187 }
188
189 // Some other item was restored at this position
190 if (!plotData.at(dataIndex).markedForDelete)
191 {
192 LOG_WARN("Adding PlotData item '{}' at position {}, but at this position exists already the item '{}'. Reordering the items to match the data. Consider resaving the flow file.",
193 displayName, dataIndex, plotData.at(dataIndex).displayName);
194 }
195 auto searchIter = std::find_if(plotData.begin(),
196 plotData.end(),
197 [displayName](const PlotData& plotData) { return plotData.displayName == displayName; });
198 auto iter = plotData.begin();
199 std::advance(iter, dataIndex);
200 iter->markedForDelete = false;
201 if (searchIter == plotData.end()) // Item does not exist yet. Developer added a new item to the list
202 {
203 plotData.insert(iter, PlotData{ displayName, static_cast<size_t>(size) });
204 }
205 else // Item exists already. Developer reordered the items in the list
206 {
207 move(plotData, static_cast<size_t>(searchIter - plotData.begin()), dataIndex);
208 }
209 }
210 else if (std::find_if(plotData.begin(),
211 plotData.end(),
212 [displayName](const PlotData& plotData) { return plotData.displayName == displayName; })
213 != plotData.end())
214 {
215 LOG_ERROR("Adding the PlotData item {} at position {}, but this plot item was found at another position already",
216 displayName, dataIndex);
217 }
218 else // Item not there yet. Add to the end of the list
219 {
220 plotData.emplace_back(displayName, static_cast<size_t>(size));
221 }
222 }
