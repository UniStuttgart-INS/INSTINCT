70 [[nodiscard]] std::optional<double> getValueAt(size_t idx) const override
71 {
72 INS_ASSERT(idx < GetStaticDescriptorCount());
73 switch (idx)
74 {
75 case 0: // Latitude [deg]
76 case 1: // Longitude [deg]
77 case 2: // Altitude [m]
78 case 3: // North/South [m]
79 case 4: // East/West [m]
80 case 5: // X-ECEF [m]
81 case 6: // Y-ECEF [m]
82 case 7: // Z-ECEF [m]
83 case 8: // Velocity norm [m/s]
84 case 9: // X velocity ECEF [m/s]
85 case 10: // Y velocity ECEF [m/s]
86 case 11: // Z velocity ECEF [m/s]
87 case 12: // North velocity [m/s]
88 case 13: // East velocity [m/s]
89 case 14: // Down velocity [m/s]
90 return PosVel::getValueAt(idx);
91 case 15: // Roll [deg]
92 return rad2deg(rollPitchYaw().x());
93 case 16: // Pitch [deg]
94 return rad2deg(rollPitchYaw().y());
95 case 17: // Yaw [deg]
96 return rad2deg(rollPitchYaw().z());
97 case 18: // Quaternion::w
98 return n_Quat_b().w();
99 case 19: // Quaternion::x
100 return n_Quat_b().x();
101 case 20: // Quaternion::y
102 return n_Quat_b().y();
103 case 21: // Quaternion::z
104 return n_Quat_b().z();
105 default:
106 return std::nullopt;
107 }
108 }
