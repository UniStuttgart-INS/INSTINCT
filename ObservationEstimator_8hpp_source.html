51 template<typename ReceiverType>
52 void calcObservationEstimates(Observations& observations,
53 const std::array<Receiver<ReceiverType>, ReceiverType::ReceiverType_COUNT>& receivers,
54 const IonosphericCorrections& ionosphericCorrections,
55 [[maybe_unused]] const std::string& nameId,
56 ObservationDifference obsDiff)
57 {
58 LOG_DATA("{}: Calculating observation estimates:", nameId);
59
60 for (auto& [satSigId, observation] : observations.signals)
61 {
62 const Frequency freq = satSigId.freq();
63 const SatelliteSystem satSys = freq.getSatSys();
64
65 for (size_t r = 0; r < observation.recvObs.size(); r++)
66 {
67 auto& recvObs = observation.recvObs.at(r);
68 [[maybe_unused]] auto recv = static_cast<ReceiverType>(r);
69 const auto& receiver = receivers.at(r);
70
71 // Receiver-Satellite Range [m]
72 double rho_r_s = (recvObs.e_satPos() - receiver.e_pos).norm();
73 recvObs.terms.rho_r_s = rho_r_s;
74 // Troposphere
75 auto tropo_r_s = calcTroposphericDelayAndMapping(receiver.gnssObs->insTime, receiver.lla_pos,
76 recvObs.satElevation(), recvObs.satAzimuth(), _troposphereModels);
77 recvObs.terms.tropoZenithDelay = tropo_r_s;
78 // Estimated troposphere propagation error [m]
79 double dpsr_T_r_s = tropo_r_s.ZHD * tropo_r_s.zhdMappingFactor + tropo_r_s.ZWD * tropo_r_s.zwdMappingFactor;
80 recvObs.terms.dpsr_T_r_s = dpsr_T_r_s;
81 // Estimated ionosphere propagation error [m]
82 double dpsr_I_r_s = calcIonosphericDelay(static_cast<double>(receiver.gnssObs->insTime.toGPSweekTow().tow),
83 freq, observation.freqNum(), receiver.lla_pos, recvObs.satElevation(), recvObs.satAzimuth(),
84 _ionosphereModel, &ionosphericCorrections);
85 recvObs.terms.dpsr_I_r_s = dpsr_I_r_s;
86 // Sagnac correction [m]
87 double dpsr_ie_r_s = calcSagnacCorrection(receiver.e_pos, recvObs.e_satPos());
88 recvObs.terms.dpsr_ie_r_s = dpsr_ie_r_s;
89
90 // Earth's gravitational field causes relativistic signal delay due to space-time curvature (Shapiro effect) [s]
91 // double posNorm = recvObs.e_satPos().norm() + receiver.e_pos.norm();
92 // double dt_rel_stc = receiver.e_pos.norm() > InsConst<>::WGS84::a / 2.0 ? 2.0 * InsConst<>::WGS84::MU / std::pow(InsConst<>::C, 3) * std::log((posNorm + rho_r_s) / (posNorm - rho_r_s))
93 // : 0.0;
94
95 double cn0 = recvObs.gnssObsData().CN0.value_or(1.0);
96
97 for (auto& [obsType, obsData] : recvObs.obs)
98 {
99 LOG_DATA("{}: [{}][{:11}][{:5}] Observation estimate", nameId, satSigId, obsType, recv);
100 switch (obsType)
101 {
102 case GnssObs::Pseudorange:
103 obsData.estimate = rho_r_s
104 + dpsr_ie_r_s
105 + dpsr_T_r_s
106 + dpsr_I_r_s
107 + InsConst<>::C
108 * (receiver.recvClk.bias.value * (obsDiff != DoubleDifference)
109 - recvObs.satClock().bias * (obsDiff == NoDifference)
110 + receiver.recvClk.sysTimeDiffBias.at(satSys.toEnumeration()).value
111 // + dt_rel_stc
112 + (receiver.interFrequencyBias.contains(freq) ? receiver.interFrequencyBias.at(freq).value : 0.0));
113 obsData.measVar = _gnssMeasurementErrorModel.psrMeasErrorVar(satSys, recvObs.satElevation(), cn0);
114 LOG_DATA("{}: [{}][{:11}][{:5}] {:.4f} [m] Geometrical range", nameId, satSigId, obsType, recv, rho_r_s);
115 LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Sagnac correction", nameId, satSigId, obsType, recv, dpsr_ie_r_s);
116 if (dpsr_T_r_s != 0.0) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Tropospheric delay", nameId, satSigId, obsType, recv, dpsr_T_r_s); }
117 if (dpsr_I_r_s != 0.0) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Ionospheric delay", nameId, satSigId, obsType, recv, dpsr_I_r_s); }
118 if (obsDiff != DoubleDifference) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Receiver clock bias", nameId, satSigId, obsType, recv, InsConst<>::C * receiver.recvClk.bias.value); }
119 if (obsDiff == NoDifference) { LOG_DATA("{}: [{}][{:11}][{:5}] - {:.4f} [m] Satellite clock bias", nameId, satSigId, obsType, recv, InsConst<>::C * recvObs.satClock().bias); }
120 if (receiver.recvClk.sysTimeDiffBias.at(satSys.toEnumeration()).value != 0.0) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Inter-system clock bias", nameId, satSigId, obsType, recv, InsConst<>::C * receiver.recvClk.sysTimeDiffBias.at(satSys.toEnumeration()).value); }
121 // LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Shapiro effect", nameId, satSigId, obsType, recv, InsConst<>::C * dt_rel_stc);
122 if (receiver.interFrequencyBias.contains(freq)) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Inter-frequency bias", nameId, satSigId, obsType, recv, InsConst<>::C * receiver.interFrequencyBias.at(freq).value); }
123 LOG_DATA("{}: [{}][{:11}][{:5}] = {:.4f} [m] Pseudorange estimate", nameId, satSigId, obsType, recv, obsData.estimate);
124 LOG_DATA("{}: [{}][{:11}][{:5}] {:.4e} [m] Difference to measurement", nameId, satSigId, obsType, recv, obsData.measurement - obsData.estimate);
125 break;
126 case GnssObs::Carrier:
127 obsData.estimate = rho_r_s
128 + dpsr_ie_r_s
129 + dpsr_T_r_s
130 - dpsr_I_r_s
131 + InsConst<>::C
132 * (receiver.recvClk.bias.value * (obsDiff != DoubleDifference)
133 - recvObs.satClock().bias * (obsDiff == NoDifference)
134 + receiver.recvClk.sysTimeDiffBias.at(satSys.toEnumeration()).value
135 // + dt_rel_stc
136 );
137 obsData.measVar = _gnssMeasurementErrorModel.carrierMeasErrorVar(satSys, recvObs.satElevation(), cn0);
138 LOG_DATA("{}: [{}][{:11}][{:5}] {:.4f} [m] Geometrical range", nameId, satSigId, obsType, recv, rho_r_s);
139 LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Sagnac correction", nameId, satSigId, obsType, recv, dpsr_ie_r_s);
140 if (dpsr_T_r_s != 0.0) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Tropospheric delay", nameId, satSigId, obsType, recv, dpsr_T_r_s); }
141 if (dpsr_I_r_s != 0.0) { LOG_DATA("{}: [{}][{:11}][{:5}] - {:.4f} [m] Ionospheric delay", nameId, satSigId, obsType, recv, dpsr_I_r_s); }
142 if (obsDiff != DoubleDifference) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Receiver clock bias", nameId, satSigId, obsType, recv, InsConst<>::C * receiver.recvClk.bias.value); }
143 if (obsDiff == NoDifference) { LOG_DATA("{}: [{}][{:11}][{:5}] - {:.4f} [m] Satellite clock bias", nameId, satSigId, obsType, recv, InsConst<>::C * recvObs.satClock().bias); }
144 if (receiver.recvClk.sysTimeDiffBias.at(satSys.toEnumeration()).value != 0.0) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Inter-system clock bias", nameId, satSigId, obsType, recv, InsConst<>::C * receiver.recvClk.sysTimeDiffBias.at(satSys.toEnumeration()).value); }
145 // LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m] Shapiro effect", nameId, satSigId, obsType, recv, InsConst<>::C * dt_rel_stc);
146 LOG_DATA("{}: [{}][{:11}][{:5}] = {:.4f} [m] Carrier-phase estimate", nameId, satSigId, obsType, recv, obsData.estimate);
147 LOG_DATA("{}: [{}][{:11}][{:5}] {:.4e} [m] Difference to measurement", nameId, satSigId, obsType, recv, obsData.measurement - obsData.estimate);
148 break;
149 case GnssObs::Doppler:
150 obsData.estimate = recvObs.e_pLOS().dot(recvObs.e_satVel() - receiver.e_vel)
151 - calcSagnacRateCorrection(receiver.e_pos, recvObs.e_satPos(), receiver.e_vel, recvObs.e_satVel())
152 + InsConst<>::C
153 * (receiver.recvClk.drift.value * (obsDiff != DoubleDifference)
154 - recvObs.satClock().drift * (obsDiff == NoDifference)
155 + receiver.recvClk.sysTimeDiffDrift.at(satSys.toEnumeration()).value * (obsDiff == NoDifference));
156 obsData.measVar = _gnssMeasurementErrorModel.psrRateMeasErrorVar(freq, observation.freqNum(), recvObs.satElevation(), cn0);
157 LOG_DATA("{}: [{}][{:11}][{:5}] {:.4f} [m/s] ", nameId, satSigId, obsType, recv, recvObs.e_pLOS().dot(recvObs.e_satVel() - receiver.e_vel));
158 LOG_DATA("{}: [{}][{:11}][{:5}] - {:.4f} [m/s] Sagnac rate correction", nameId, satSigId, obsType, recv, calcSagnacRateCorrection(receiver.e_pos, recvObs.e_satPos(), receiver.e_vel, recvObs.e_satVel()));
159 if (obsDiff != DoubleDifference) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m/s] Receiver clock drift", nameId, satSigId, obsType, recv, InsConst<>::C * receiver.recvClk.drift.value); }
160 if (obsDiff == NoDifference) { LOG_DATA("{}: [{}][{:11}][{:5}] - {:.4f} [m/s] Satellite clock drift", nameId, satSigId, obsType, recv, InsConst<>::C * recvObs.satClock().drift); }
161 if (receiver.recvClk.sysTimeDiffDrift.at(satSys.toEnumeration()).value != 0.0) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m/s] Inter-system clock drift", nameId, satSigId, obsType, recv, InsConst<>::C * receiver.recvClk.sysTimeDiffDrift.at(satSys.toEnumeration()).value); }
162 LOG_DATA("{}: [{}][{:11}][{:5}] = {:.4f} [m/s] Doppler estimate", nameId, satSigId, obsType, recv, obsData.estimate);
163 LOG_DATA("{}: [{}][{:11}][{:5}] {:.4e} [m/s] Difference to measurement", nameId, satSigId, obsType, recv, obsData.measurement - obsData.estimate);
164 break;
165 case GnssObs::ObservationType_COUNT:
166 break;
167 }
168 LOG_DATA("{}: [{}][{:11}][{:5}] Observation error variance", nameId, satSigId, obsType, recv);
169 LOG_DATA("{}: [{}][{:11}][{:5}] {:.4g} [{}] Measurement error variance", nameId, satSigId, obsType, recv, obsData.measVar,
170 obsType == GnssObs::Doppler ? "m^2/s^2" : "m^2");
171
172 if (obsDiff == NoDifference)
173 {
174 if (obsType == GnssObs::Pseudorange || obsType == GnssObs::Carrier)
175 {
176 obsData.measVar += observation.navData()->calcSatellitePositionVariance()
177 + ionoErrorVar(dpsr_I_r_s, freq, observation.freqNum())
178 + tropoErrorVar(dpsr_T_r_s, recvObs.satElevation());
179 LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m^2] Satellite position variance", nameId, satSigId, obsType, recv, observation.navData()->calcSatellitePositionVariance());
180 if (dpsr_I_r_s != 0.0) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m^2] Ionosphere variance", nameId, satSigId, obsType, recv, ionoErrorVar(dpsr_I_r_s, freq, observation.freqNum())); }
181 if (dpsr_T_r_s != 0.0) { LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m^2] Troposphere variance", nameId, satSigId, obsType, recv, tropoErrorVar(dpsr_T_r_s, recvObs.satElevation())); }
182 }
183 if (obsType == GnssObs::Pseudorange)
184 {
185 obsData.measVar += _gnssMeasurementErrorModel.codeBiasErrorVar();
186 LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m^2] Code bias variance", nameId, satSigId, obsType, recv, _gnssMeasurementErrorModel.codeBiasErrorVar());
187
188 if (receiver.interFrequencyBias.contains(freq))
189 {
190 obsData.measVar += std::pow(receiver.interFrequencyBias.at(freq).stdDev, 2);
191 LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m^2] Inter-frequency bias variance", nameId, satSigId, obsType, recv,
192 std::pow(InsConst<>::C * receiver.interFrequencyBias.at(freq).stdDev, 2));
193 }
194 }
195 }
196 if (obsDiff != DoubleDifference)
197 {
198 if (obsType == GnssObs::Pseudorange || obsType == GnssObs::Carrier)
199 {
200 double recvClockVariance = std::pow(InsConst<>::C, 2)
201 * (std::pow(receiver.recvClk.bias.stdDev, 2)
202 + std::pow(receiver.recvClk.sysTimeDiffBias.at(satSys.toEnumeration()).stdDev, 2));
203 obsData.measVar += recvClockVariance;
204 LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m^2] Receiver clock bias variance", nameId, satSigId, obsType, recv, recvClockVariance);
205 }
206 else if (obsType == GnssObs::Doppler)
207 {
208 double recvClockVariance = std::pow(InsConst<>::C, 2)
209 * (std::pow(receiver.recvClk.drift.stdDev, 2)
210 + std::pow(receiver.recvClk.sysTimeDiffDrift.at(satSys.toEnumeration()).stdDev, 2));
211 obsData.measVar += recvClockVariance;
212 LOG_DATA("{}: [{}][{:11}][{:5}] + {:.4f} [m^2/s^2] Receiver clock drift variance", nameId, satSigId, obsType, recv, recvClockVariance);
213 }
214 }
215 LOG_DATA("{}: [{}][{:11}][{:5}] = {:.4g} [{}] Observation error variance", nameId, satSigId, obsType, recv, obsData.measVar,
216 obsType == GnssObs::Doppler ? "m^2/s^2" : "m^2");
217 }
218 }
219 }

225 bool ShowGuiWidgets(const char* id, float itemWidth)
226 {
227 bool changed = false;
228
229 ImGui::SetNextItemOpen(true, ImGuiCond_FirstUseEver);
230 if (ImGui::TreeNode(fmt::format("Compensation models##{}", id).c_str()))
231 {
232 ImGui::SetNextItemWidth(itemWidth - ImGui::GetStyle().IndentSpacing);
233 if (ComboIonosphereModel(fmt::format("Ionosphere Model##{}", id).c_str(), _ionosphereModel))
234 {
235 LOG_DEBUG("{}: Ionosphere Model changed to {}", id, NAV::to_string(_ionosphereModel));
236 changed = true;
237 }
238 if (ComboTroposphereModel(fmt::format("Troposphere Model##{}", id).c_str(), _troposphereModels, itemWidth - ImGui::GetStyle().IndentSpacing))
239 {
240 changed = true;
241 }
242 ImGui::TreePop();
243 }
244
245 ImGui::SetNextItemOpen(true, ImGuiCond_FirstUseEver);
246 if (ImGui::TreeNode(fmt::format("GNSS Measurement Error##{}", id).c_str()))
247 {
248 if (_gnssMeasurementErrorModel.ShowGuiWidgets(id, itemWidth - ImGui::GetStyle().IndentSpacing))
249 {
250 LOG_DEBUG("{}: GNSS Measurement Error Model changed.", id);
251 changed = true;
252 }
253 ImGui::TreePop();
254 }
255
256 return changed;
