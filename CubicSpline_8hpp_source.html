78 void setPoints(const std::vector<Scalar>& x, const std::vector<Scalar>& y)
79 {
80 vals_x = x;
81 vals_y = y;
82 size_t n = x.size();
83
84 size_t n_upper = (boundaryConditionLeft.type == BoundaryCondition::NotAKnot) ? 2 : 1;
85 size_t n_lower = (boundaryConditionRight.type == BoundaryCondition::NotAKnot) ? 2 : 1;
86 BandMatrix A(n, n_upper, n_lower);
87 std::vector<Scalar> rhs(n);
88 for (size_t i = 1; i < n - 1; i++)
89 {
90 A(i, i - 1) = 1.0 / 3.0 * (x[i] - x[i - 1]);
91 A(i, i) = 2.0 / 3.0 * (x[i + 1] - x[i - 1]);
92 A(i, i + 1) = 1.0 / 3.0 * (x[i + 1] - x[i]);
93 rhs[i] = (y[i + 1] - y[i]) / (x[i + 1] - x[i]) - (y[i] - y[i - 1]) / (x[i] - x[i - 1]);
94 }
95 // boundary conditions
96 if (boundaryConditionLeft.type == BoundaryCondition::SecondDerivative)
97 {
98 // 2*c[0] = f''
99 A(0, 0) = 2.0;
100 A(0, 1) = 0.0;
101 rhs[0] = boundaryConditionLeft.value;
102 }
103 else if (boundaryConditionLeft.type == BoundaryCondition::FirstDerivative)
104 {
105 // b[0] = f', needs to be re-expressed in terms of c:
106 // (2c[0]+c[1])(x[1]-x[0]) = 3 ((y[1]-y[0])/(x[1]-x[0]) - f')
107 A(0, 0) = 2.0 * (x[1] - x[0]);
108 A(0, 1) = 1.0 * (x[1] - x[0]);
109 rhs[0] = 3.0 * ((y[1] - y[0]) / (x[1] - x[0]) - boundaryConditionLeft.value);
110 }
111 else if (boundaryConditionLeft.type == BoundaryCondition::NotAKnot)
112 {
113 // f'''(x[1]) exists, i.e. d[0]=d[1], or re-expressed in c:
114 // -h1*c[0] + (h0+h1)*c[1] - h0*c[2] = 0
115 A(0, 0) = -(x[2] - x[1]);
116 A(0, 1) = x[2] - x[0];
117 A(0, 2) = -(x[1] - x[0]);
118 rhs[0] = 0.0;
119 }
120 if (boundaryConditionRight.type == BoundaryCondition::SecondDerivative)
121 {
122 // 2*c[n-1] = f''
123 A(n - 1, n - 1) = 2.0;
124 A(n - 1, n - 2) = 0.0;
125 rhs[n - 1] = boundaryConditionRight.value;
126 }
127 else if (boundaryConditionRight.type == BoundaryCondition::FirstDerivative)
128 {
129 // b[n-1] = f', needs to be re-expressed in terms of c:
130 // (c[n-2]+2c[n-1])(x[n-1]-x[n-2])
131 // = 3 (f' - (y[n-1]-y[n-2])/(x[n-1]-x[n-2]))
132 A(n - 1, n - 1) = 2.0 * (x[n - 1] - x[n - 2]);
133 A(n - 1, n - 2) = 1.0 * (x[n - 1] - x[n - 2]);
134 rhs[n - 1] = 3.0 * (boundaryConditionRight.value - (y[n - 1] - y[n - 2]) / (x[n - 1] - x[n - 2]));
135 }
136 else if (boundaryConditionRight.type == BoundaryCondition::NotAKnot)
137 {
138 // f'''(x[n-2]) exists, i.e. d[n-3]=d[n-2], or re-expressed in c:
139 // -h_{n-2}*c[n-3] + (h_{n-3}+h_{n-2})*c[n-2] - h_{n-3}*c[n-1] = 0
140 A(n - 1, n - 3) = -(x[n - 1] - x[n - 2]);
141 A(n - 1, n - 2) = x[n - 1] - x[n - 3];
142 A(n - 1, n - 1) = -(x[n - 2] - x[n - 3]);
143 rhs[0] = 0.0;
144 }
145
146 // solve the equation system to obtain the parameters c[]
147 coef_c = A.lu_solve(rhs);
148
149 // calculate parameters b[] and d[] based on c[]
150 coef_d.resize(n);
151 coef_b.resize(n);
152 for (size_t i = 0; i < n - 1; i++)
153 {
154 coef_d[i] = 1.0 / 3.0 * (coef_c[i + 1] - coef_c[i]) / (x[i + 1] - x[i]);
155 coef_b[i] = (y[i + 1] - y[i]) / (x[i + 1] - x[i])
156 - 1.0 / 3.0 * (2.0 * coef_c[i] + coef_c[i + 1]) * (x[i + 1] - x[i]);
157 }
158 // for the right extrapolation coefficients (zero cubic term)
159 // f_{n-1}(x) = y_{n-1} + b*(x-x_{n-1}) + c*(x-x_{n-1})^2
160 auto h = x[n - 1] - x[n - 2];
161 // coef_c[n-1] is determined by the boundary condition
162 coef_d[n - 1] = 0.0;
163 coef_b[n - 1] = 3.0 * coef_d[n - 2] * h * h + 2.0 * coef_c[n - 2] * h + coef_b[n - 2]; // = f'_{n-2}(x_{n-1})
164 if (boundaryConditionRight.type == BoundaryCondition::FirstDerivative)
165 {
166 coef_c[n - 1] = 0.0; // force linear extrapolation
167 }
168 // for left extrapolation coefficients
169 coef_c0 = (boundaryConditionLeft.type == BoundaryCondition::FirstDerivative) ? 0.0 : coef_c[0];
170 }

181 [[nodiscard]] Scalar operator()(Scalar x) const
182 {
183 size_t n = vals_x.size();
184 size_t idx = findClosestIdx(x);
185
186 auto h = x - vals_x[idx];
187
188 if (x < vals_x[0]) // extrapolation to the left
189 {
190 return (coef_c0 * h + coef_b[0]) * h + vals_y[0];
191 }
192 if (x > vals_x[n - 1]) // extrapolation to the right
193 {
194 return (coef_c[n - 1] * h + coef_b[n - 1]) * h + vals_y[n - 1];
195 }
196 // interpolation
197 return ((coef_d[idx] * h + coef_c[idx]) * h + coef_b[idx]) * h + vals_y[idx];
198 }

204 [[nodiscard]] Scalar derivative(size_t order, Scalar x) const
205 {
206 size_t n = vals_x.size();
207 size_t idx = findClosestIdx(x);
208
209 auto h = x - vals_x[idx];
210 if (x < vals_x[0]) // extrapolation to the left
211 {
212 switch (order)
213 {
214 case 1:
215 return 2.0 * coef_c0 * h + coef_b[0];
216 case 2:
217 return 2.0 * coef_c0;
218 default:
219 return 0.0;
220 }
221 }
222 else if (x > vals_x[n - 1]) // extrapolation to the right
223 {
224 switch (order)
225 {
226 case 1:
227 return 2.0 * coef_c[n - 1] * h + coef_b[n - 1];
228 case 2:
229 return 2.0 * coef_c[n - 1];
230 default:
231 return 0.0;
232 }
233 }
234 else // interpolation
235 {
236 switch (order)
237 {
238 case 1:
239 return (3.0 * coef_d[idx] * h + 2.0 * coef_c[idx]) * h + coef_b[idx];
240 case 2:
241 return 6.0 * coef_d[idx] * h + 2.0 * coef_c[idx];
242 case 3:
243 return 6.0 * coef_d[idx];
244 default:
245 return 0.0;
246 }
247 }
248 }

297 [[nodiscard]] std::vector<Scalar> lu_solve(const std::vector<Scalar>& b, bool is_lu_decomposed = false)
298 {
299 INS_ASSERT_USER_ERROR(dim() == b.size(), "The BandMatrix dimension must be the same as the vector b in the equation Ax = b.");
300
301 if (!is_lu_decomposed)
302 {
303 lu_decompose();
304 }
305 auto y = l_solve(b);
306 auto x = u_solve(y);
307 return x;
308 }

312 [[nodiscard]] size_t dim() const
313 {
314 return !upperBand.empty() ? upperBand[0].size() : 0;
315 }
317 [[nodiscard]] size_t dimUpperBand() const
318 {
319 return upperBand.size() - 1;
320 }
322 [[nodiscard]] size_t dimLowerBand() const
323 {
324 return lowerBand.size() - 1;

328 void lu_decompose()
329 {
330 // preconditioning
331 // normalize column i so that a_ii=1
332 for (size_t i = 0; i < dim(); i++)
333 {
334 assert(this->operator()(i, i) != 0.0);
335 lowerBand[0][i] = 1.0 / this->operator()(i, i);
336 size_t j_min = i > dimLowerBand() ? i - dimLowerBand() : 0;
337 size_t j_max = std::min(dim() - 1, i + dimUpperBand());
338 for (size_t j = j_min; j <= j_max; j++)
339 {
340 this->operator()(i, j) *= lowerBand[0][i];
341 }
342 this->operator()(i, i) = 1.0; // prevents rounding errors
343 }
344
345 // Gauss LR-Decomposition
346 for (size_t k = 0; k < dim(); k++)
347 {
348 size_t i_max = std::min(dim() - 1, k + dimLowerBand());
349 for (size_t i = k + 1; i <= i_max; i++)
350 {
351 assert(this->operator()(k, k) != 0.0);
352
353 auto x = -this->operator()(i, k) / this->operator()(k, k);
354 this->operator()(i, k) = -x; // assembly part of L
355 size_t j_max = std::min(dim() - 1, k + dimUpperBand());
356 for (size_t j = k + 1; j <= j_max; j++)
357 {
358 // assembly part of R
359 this->operator()(i, j) = this->operator()(i, j) + x * this->operator()(k, j);
360 }
361 }
362 }
363 }
364
366 [[nodiscard]] std::vector<Scalar> l_solve(const std::vector<Scalar>& b) const
367 {
368 INS_ASSERT_USER_ERROR(dim() == b.size(), "The BandMatrix dimension must be the same as the vector b in the equation Ax = b.");
369
370 std::vector<Scalar> x(dim());
371 for (size_t i = 0; i < dim(); i++)
372 {
373 Scalar sum = 0;
374 size_t j_start = i > dimLowerBand() ? i - dimLowerBand() : 0;
375 for (size_t j = j_start; j < i; j++)
376 {
377 sum += this->operator()(i, j) * x[j];
378 }
379 x[i] = (b[i] * lowerBand[0][i]) - sum;
380 }
381 return x;
382 }
384 [[nodiscard]] std::vector<Scalar> u_solve(const std::vector<Scalar>& b) const
385 {
386 INS_ASSERT_USER_ERROR(dim() == b.size(), "The BandMatrix dimension must be the same as the vector b in the equation Ax = b.");
387
388 std::vector<Scalar> x(dim());
389 for (size_t i = dim(); i-- > 0;)
390 {
391 Scalar sum = 0;
392 size_t j_stop = std::min(dim() - 1, i + dimUpperBand());
393 for (size_t j = i + 1; j <= j_stop; j++)
394 {
395 sum += this->operator()(i, j) * x[j];
396 }
397 x[i] = (b[i] - sum) / this->operator()(i, i);
398 }
399 return x;
400 }

402 std::vector<std::vector<Scalar>> upperBand;
403 std::vector<std::vector<Scalar>> lowerBand;
404 };
